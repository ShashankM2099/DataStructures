public boolean isConnected() {

            Vertex rootD = this.list.get(0);

            visited(rootD);
            int unmarked = 0;

            for (Vertex element : this.list) {
                if (element.getMarked() == false) {
                    unmarked++;
                }
            }

            if (unmarked > 0) {
                return false;
            } else {
                return true;
            }
        }
